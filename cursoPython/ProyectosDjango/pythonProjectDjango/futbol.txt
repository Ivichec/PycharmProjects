create table EQUIPOS

  NOMBRE      VARCHAR2(50),

  PRESIDENTE  VARCHAR2(50),

  pctfree 10
  pctused 40
  initrans 1
  maxtrans 255
  storage
  (
    initial 64K
    next 1M
    minextents 1
    maxextents unlimited
  );
